participant GUI
participant Steuerung
participant Highscore

activate Benutzer
activate GUI
Benutzer -> GUI: Spiel starten
GUI -> Steuerung: anzeigenHighscore()
activate Steuerung

Steuerung -> Highscore: getHighscore()
activate Highscore
Highscore --> Steuerung: int highscore
deactivate Highscore

Steuerung ->Steuerung:Integer.toString(highscore)
activate Steuerung

deactivate Steuerung

activate GUI
GUI -->> Steuerung: 
deactivate GUI

Steuerung -> GUI:textAlign(dieGUI.CENTER)
activate GUI
GUI -->> Steuerung: 
deactivate GUI

Steuerung -> GUI: text(highscoreString, 100, 40)
activate GUI
GUI -->> Steuerung: 
deactivate GUI

Steuerung -->> GUI: 
deactivate Steuerung
