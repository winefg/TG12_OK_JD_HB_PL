title Spieler anmelden

actor Benutzer
participant GUI
participant MyJDBC
participant Datenbank
participant Steuerung

activate Benutzer
Benutzer->GUI:keyTyped()
activate GUI

opt focusedField == 2 && key == ENTER

GUI->MyJDBC:login(nickname, password)
activate MyJDBC
MyJDBC->Datenbank:executeQuery()
activate Datenbank
MyJDBC<<--Datenbank:executeQuery()
deactivate Datenbank
GUI<<--MyJDBC:login(nickname, password):id
deactivate MyJDBC

alt id != -1
GUI->Steuerung:setAktSpielerID(id)
activate Steuerung
GUI<<--Steuerung:setAktSpielerID(id)
deactivate Steuerung
GUI->Steuerung:getAktuellesSpiel()
activate Steuerung
GUI<<--Steuerung:getAktuellesSpiel():snakeSpiel
deactivate Steuerung
GUI->Steuerung:addSpiel(snakeSpiel)
activate Steuerung
GUI<<--Steuerung:addSpiel(snakeSpiel)
deactivate Steuerung
Benutzer<<--GUI:Login erfolgreich
deactivate GUI
else
activate GUI
Benutzer<<--GUI:println("Wrong login")
deactivate GUI
end
end
