void drawSpielFeld() {
  background(#B3D665);

  int cellSize = 40;
  int gridSize = 19;

  int fieldWidth = gridSize * cellSize;
  int fieldHeight = gridSize * cellSize;

  // Anfang des Feldes
  int startX = (width - fieldWidth) / 2;   // in der Mitte (X-Achse)
  int startY = 120;                        // oben Platz für Highscore machen

  fill(#ACD05E);
  noStroke();

  for (int y = 0; y < gridSize; y++) {
    for (int x = 0; x < gridSize; x++) {

      if ((x + y) % 2 == 1) {
        rect(
          startX + x * cellSize,
          startY + y * cellSize,
          cellSize,
          cellSize
        );
      }
    }
  }
}
